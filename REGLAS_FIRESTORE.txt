rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    match /registros/{id} {
      allow create: if true;
      allow read, update, delete: if request.auth != null;
    }

    match /config/{id} {
      allow read: if true;

      allow create, delete: if request.auth != null;

      allow update: if request.auth != null
        || (
          request.resource.data.diff(resource.data).changedKeys().hasOnly(['registrados', 'updatedAt'])
          && request.resource.data.registrados == resource.data.registrados + 1
          && resource.data.activo == true
          && resource.data.registrados < resource.data.cupos
        );
    }
  }
}
